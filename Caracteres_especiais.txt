======Caracteres especiais======


<!-- Aspas duplas -->
<p>&quot;</p>
----------------------------------
<!-- Aspas simples -->
<p>&apos;</p>
----------------------------------
<!-- Menor que -->
<p>&lt;</p>
----------------------------------
<!-- Maior que -->
<p>&gt;</p>
----------------------------------
<!-- E comercial -->
<p>&amp;</p>
----------------------------------
<!-- Espaço em branco fixo -->
<p>Espaço&nbsp;entre</p>
----------------------------------
<!-- Direitos autorais -->
<p>&copy; 2025</p>
----------------------------------
<!-- Marca registrada -->
<p>Marca&reg;</p>
----------------------------------
<!-- Símbolo do Euro -->
<p>&euro;100</p>
----------------------------------
<!-- Centavos -->
<p>50&cent;</p>
----------------------------------
<!-- Iene japonês -->
<p>&yen;500</p>
----------------------------------
<!-- Reticências -->
<p>Espere&hellip;</p>
----------------------------------
<!-- Parágrafo -->
<p>&para;</p>
----------------------------------
<!-- Marca comercial -->
<p>Produto&trade;</p>
----------------------------------
<!-- Traço médio -->
<p>2020&ndash;2025</p>
----------------------------------
<!-- Travessão -->
<p>Texto&mdash;com travessão</p>
----------------------------------
<!-- Exclamação invertida -->
<p>&iexcl;Hola!</p>
----------------------------------
<!-- Interrogação invertida -->
<p>&iquest;Qué tal?</p>
----------------------------------
<!-- Cerquilha (hashtag) -->
<p>&#35;hashtag</p>
----------------------------------
